const inputs = document.querySelectorAll('input'); const patterns ={ username : /^[a-z\d]{3,20}$/i, email : /^([a-z\d\.-]+)@([a-z\d-]+)\.([a-z]{2,8})(\.[a-z]{2,8})?$/, password : /^(?=.*[0-9])(?=.*[!@#$%^&*])[a-zA-Z0-9!@#$%^&*]{6,16}$/, telephone : /^\d{10}$/ }; function validate(field,regex){ if(regex.test(field.value)){ field.className="valid"; } else{ field.className="invalid"; } } inputs.forEach((input)=>{ input.addEventListener('keyup',(e)=>{ validate(e.target,patterns[e.target.attributes.name.value]); }) })
